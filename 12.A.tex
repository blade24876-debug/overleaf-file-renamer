\subsection*{12.A.1 Template Purpose and Use Rules — Operational}
\phantomsection
\addcontentsline{toc}{subsection}{12.A.1 Template Purpose and Use Rules — Operational}

\begin{itemize}
\item This appendix implements Stage 2.E minimum datasets and Stage 1.F validity doctrine.
\item Missing required fields make the record inadmissible for gate decisions.
\item Do not infer or reconstruct missing values; record \textbf{UNKNOWN} or \textbf{MISSING}.
\item Session and test validity tags must reflect completion and documentation, not effort.
\item Any late entry must be verifiable; unverifiable late entry remains \textbf{MISSING} and the record remains inadmissible for gate use.
\end{itemize}

\subsection*{12.A.2 Identifier Header — Required on Every Log Page}
\phantomsection
\addcontentsline{toc}{subsection}{12.A.2 Identifier Header — Required on Every Log Page}

\begin{itemize}
\item Program Version:
\item Phase:
\item Phase Week:
\item Session \#:
\item Timestamp ISO 8601 with offset:
\item Location:
\item Environment Unit \textbf{ID} (\textbf{ROUTE}-\#\# / \textbf{POOL}-\#\# / \textbf{MACH}-\#\#):
\item Notes:
\item \textbf{ISS-08}-\#\#\# reference if a session numbering conflict is detected:
\end{itemize}

Rules:

\begin{itemize}
\item Phase must be Phase 00 through Phase 13.
\item Phase Week must be Wk \#\# format.
\item Session \# must be 1–6.
\item Environment Unit \textbf{ID} must be one of \textbf{ROUTE}-\#\#, \textbf{POOL}-\#\#, or \textbf{MACH}-\#\# as applicable, recorded in a single combined field.
\end{itemize}

\subsection*{12.A.3 Reason Code Mini-Legend — Embedded}
\phantomsection
\addcontentsline{toc}{subsection}{12.A.3 Reason Code Mini-Legend — Embedded}

\begin{itemize}
\item \textbf{ENV} — environment constraint or unsafe conditions
\item PAIN — pain or mechanics alteration
\item \textbf{EQUIP} — equipment limitation or failure
\item \textbf{TIME} — time or schedule constraint
\item \textbf{ILL} — illness or acute symptoms posture, non-diagnostic
\item \textbf{SAFETY} — safety stop or red-flag posture
\end{itemize}

Rules:

\begin{itemize}
\item Session reason codes are recorded in \textbf{SESSION-RC}.
\item Test invalidation reason codes are recorded in \textbf{TEST-RC}.
\item Free-text notes must accompany any populated reason code.
\end{itemize}

Rules block:

\begin{itemize}
\item \textbf{VALID} requires all mandatory session elements completed in order:
  \begin{enumerate}
  \item Safety self-check
  \item Warm-up
  \item Mobility
  \item Main work
  \item Cardio
  \item Cooldown
  \item Recovery notes
  \item Post-session notes
  \end{enumerate}
\item \textbf{MODIFIED} requires:
  \begin{itemize}
  \item all mandatory elements completed
  \item risk-reducing substitution documented
  \item \textbf{SESSION-RC} populated
  \end{itemize}
\item \textbf{INVALID} requires:
  \begin{itemize}
  \item any mandatory element missing
  \item or undocumented deviation
  \item or required logging fields missing
  \end{itemize}
\item \textbf{SESSION-RC}:
  \begin{itemize}
  \item blank or N/A when \textbf{VALID}
  \item required when \textbf{MODIFIED} or \textbf{INVALID}
  \end{itemize}
\item Main Work Domains must use Stage 2 domain tags only:
  \begin{itemize}
  \item \textbf{RUN}, \textbf{SWIM}, \textbf{CAL}, \textbf{RUCK}, \textbf{STR}, \textbf{BODYCOMP}, \textbf{DURABILITY}, \textbf{RECOVERY}, plus \textbf{AER} if applicable per Stage 2
  \end{itemize}
\end{itemize}

  \begin{itemize}

  \end{itemize}

\item Evidence packet pointers (binder pages or folder paths):
\item Notes and next-step actions:
\end{itemize}

Rules:

\begin{itemize}
\item Gate recommendation must be one of \textbf{ADVANCE}, \textbf{HOLD}, \textbf{REGRESS}, \textbf{MEDICAL} \textbf{HOLD} only.
\item If an accepted known issue influenced the gate recommendation, list the relevant \textbf{ISS-08}-\#\#\# in Issue IDs referenced.
\item Evidence packet pointers must reference binder page ranges and/or package folder paths only.
\end{itemize}
