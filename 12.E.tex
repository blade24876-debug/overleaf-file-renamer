\subsection*{12.E.1 Sustainment Purpose and Non-Redesign Boundary}
\phantomsection
\addcontentsline{toc}{subsection}{12.E.1 Sustainment Purpose and Non-Redesign Boundary}

\begin{itemize}
\item This appendix transitions Program v1.0 execution into sustainment posture without adding new training content, new tests, new thresholds, or phase edits.
\item Sustainment preserves:
  \begin{itemize}
  \item validity and logging discipline (Stage 1.F tags; Stage 2.E required fields),
  \item safety governance and stop posture (\textbf{STOP} \textbf{NOW} / \textbf{MEDICAL} \textbf{HOLD} posture retained),
  \item version and archive discipline (Stage 9 packaging/versioning doctrine; canonical versioning location preserved).
  \end{itemize}
\item Sustainment decisions \textbf{MUST} NOT use inadmissible evidence:
  \begin{itemize}
  \item record \textbf{UNKNOWN} / \textbf{MISSING}; no inference,
  \item \textbf{INVALID} tests provide no gate credit and are treated as not yet tested for readiness verification,
  \item reslot discipline applies; no stacking make-ups.
  \end{itemize}
\end{itemize}
